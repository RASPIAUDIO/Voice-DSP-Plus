# The goal is to produce a single variable BUILD_PROFILES containing a list of all valid build profile names,
#     BUILD_PROFILES == "intdev-lr16-lin-spi;ua-io48-sqr; [...]"

# Each build profile will automagically then have a BUILD_OPTS variable listing all compile definitions for that build profile created
#     BUILD_OPTS_intdev-lr16-lin-spi == "INT_DEVICE=1;appconfLRCLK_NOMINAL_HZ=16000;[...]"

# This is done by the following build name parser.

# Full build profile name is currently
#     DEVICE_CONFIG[-I2S_RATE][-USB_SAMPLE_RATE]-MIC_GEOMETRY[-CONTROL_TRANSPORT]
# Special build names, such as
#     demo
# are permitted; custom logic must be written for each of these as required.

macro(add_to_build_opts OPTION_STRING)
    list(APPEND BUILD_OPTS_${PROFILE} ${OPTION_STRING})
endmacro()

macro(add_to_build_sources OPTION_STRING)
    list(APPEND BUILD_SOURCES_${PROFILE} ${OPTION_STRING})
endmacro()
macro(add_to_build_libs OPTION_STRING)
    list(APPEND BUILD_LIBS_${PROFILE} ${OPTION_STRING})
endmacro()

        COMMAND ${Python3_EXECUTABLE} -m ${YAML_PROC_MULTI} -i ${YAML_CONTROL_COMMANDS} -t ${YAML_TEMPLATES_CMD}/host_cmd_map_impl.h.jinja -o ${HOST_CMD_OUT} --remove-internal
        COMMENT "Generating host_cmd_map.h"
        DEPENDS ${YAML_CONTROL_COMMANDS} ${YAML_TEMPLATES_CMD}/host_cmd_map_impl.h.jinja
        VERBATIM
    )
    add_custom_command(
        OUTPUT ${RANGE_CHECK_OUT}
        COMMAND ${Python3_EXECUTABLE} -m ${YAML_PROC_MULTI} -i ${YAML_CONTROL_COMMANDS} -t ${YAML_TEMPLATES_CMD}/host_range_check_impl.h.jinja -o ${RANGE_CHECK_OUT} --remove-internal
        COMMENT "Generating range_check.h"
        DEPENDS ${YAML_CONTROL_COMMANDS} ${YAML_TEMPLATES_CMD}/host_range_check_impl.h.jinja
        VERBATIM
    )
    add_custom_command(
        OUTPUT ${RANGE_MAP_OUT}
        COMMAND ${Python3_EXECUTABLE} -m ${YAML_PROC_MULTI} -i ${YAML_CONTROL_COMMANDS} -t ${YAML_TEMPLATES_CMD}/host_range_map_impl.h.jinja -o ${RANGE_MAP_OUT} --remove-internal
        COMMENT "Generating range_map.h"
        DEPENDS ${YAML_CONTROL_COMMANDS} ${YAML_TEMPLATES_CMD}/host_range_map_impl.h.jinja
        VERBATIM
    )
    add_custom_command(
        OUTPUT ${PRINT_ENUM_FUNC_OUT}
        COMMAND ${Python3_EXECUTABLE} -m ${YAML_PROC_ENUM} -i ${YAML_CONTROL_COMMANDS} -t ${YAML_TEMPLATES_ENUM}/print_enum_functions.h.jinja -o ${PRINT_ENUM_FUNC_OUT} --remove-internal -e ${YAML_SETTINGS_DIR}/device_enums.yaml
        COMMENT "Generating print_enum_functions.h"
        DEPENDS ${YAML_CONTROL_COMMANDS} ${YAML_TEMPLATES_ENUM}/print_enum_functions.h.jinja ${YAML_SETTINGS_DIR}/device_enums.yaml
        VERBATIM
    )
    add_custom_command(
        OUTPUT ${SUPER_PRINT_ARG_OUT}
        COMMAND ${Python3_EXECUTABLE} -m ${YAML_PROC_ENUM} -i ${YAML_CONTROL_COMMANDS} -t ${YAML_TEMPLATES_ENUM}/super_print_arg.h.jinja -o ${SUPER_PRINT_ARG_OUT} --remove-internal -e ${YAML_SETTINGS_DIR}/device_enums.yaml
        COMMENT "Generating super_print_arg.h"

    add_custom_command(

        COMMENT "Generating enum_str_map.h"
        DEPENDS ${YAML_SETTINGS_DIR}/device_enums.yaml ${YAML_TEMPLATES_ENUM}/enum_str_map.h.jinja

    list(APPEND YAML_GENERATED_FILES ${HOST_CMD_OUT} ${RANGE_CHECK_OUT} ${RANGE_MAP_OUT} ${PRINT_ENUM_FUNC_OUT} ${SUPER_PRINT_ARG_OUT} ${ENUM_STR_MAP_OUT})
    if(NOT TARGET yaml_generation_${PRODUCT_DEFAULT_SPECIFIER})
        add_custom_target(yaml_generation_${PRODUCT_DEFAULT_SPECIFIER}
            DEPENDS ${YAML_GENERATED_FILES})
    endif()
    # Create includes list for each build profile
    set(APP_INCLUDES_${PROFILE} ${APP_INCLUDES})
    # Add files with default values
    list(APPEND APP_INCLUDES_${PROFILE} ${AUTOGEN_DIR}/common ${AUTOGEN_DIR}/device ${AUTOGEN_DIR}/device/${PRODUCT_DEFAULT_SPECIFIER} ${AUTOGEN_DIR}/common/${PRODUCT_DEFAULT_SPECIFIER})

endforeach()

foreach(BUILD_PROFILE ${BUILD_PROFILES})

    set(APP_COMPILE_DEFINITIONS ${APP_COMPILE_DEFINITIONS_COMMON})
    list(APPEND APP_COMPILE_DEFINITIONS ${BUILD_OPTS_${BUILD_PROFILE}})
    set(APP_SOURCES ${APP_SOURCES_COMMON})
    list(APPEND APP_SOURCES ${BUILD_SOURCES_${BUILD_PROFILE}})
    set(APP_LINK_LIBRARIES ${APP_COMMON_LINK_LIBRARIES})
    list(APPEND APP_LINK_LIBRARIES ${BUILD_LIBS_${BUILD_PROFILE}})

    set(TARGET_SUFFIX "${BUILD_PROFILE}")

    #**********************
    # Tile Targets
    #**********************
    set(LIB_NAME "lib_application_xvf3800_${TARGET_SUFFIX}")
    add_library(${LIB_NAME} INTERFACE EXCLUDE_FROM_ALL)

    add_dependencies(${LIB_NAME} yaml_generation_${PRODUCT_DEFAULT_SPECIFIER_${BUILD_PROFILE}})

    target_sources(
        ${LIB_NAME}
        INTERFACE
        ${APP_SOURCES}
    )
    target_include_directories(
        ${LIB_NAME}
        INTERFACE
        ${APP_INCLUDES_${BUILD_PROFILE}}
    )

    target_compile_definitions(${LIB_NAME} INTERFACE ${APP_COMPILE_DEFINITIONS})
    target_compile_options(${LIB_NAME} INTERFACE ${APP_COMPILER_FLAGS})
    target_link_libraries(${LIB_NAME} INTERFACE ${APP_LINK_LIBRARIES})
    target_link_options(${LIB_NAME} INTERFACE ${APP_LINK_OPTIONS})

    foreach(TILE 0 1)
        set(TARGET_NAME "tile${TILE}_application_xvf3800_${TARGET_SUFFIX}")
        set(TILE_LIB "lib_tile${TILE}_application_xvf3800_${TARGET_SUFFIX}")
        add_executable(${TARGET_NAME} EXCLUDE_FROM_ALL src/main_tile${TILE}.c)
        add_library(${TILE_LIB} INTERFACE)

        target_compile_definitions(${TILE_LIB} INTERFACE THIS_XCORE_TILE=${TILE})
        target_link_libraries(${TILE_LIB} INTERFACE ${LIB_NAME})
        # Link xshf::xshf library here, so that the app uses lib_xcore_math from fwk_xvf::xvf
        target_link_libraries(${TILE_LIB} INTERFACE xshf::xshf)
        if(TILE EQUAL 1)
            target_link_libraries(${TILE_LIB} INTERFACE fwk_xvf::ep_proxy_remote)
        else()
            target_link_libraries(${TILE_LIB} INTERFACE fwk_xvf::ep_proxy_local)
        endif()

        # Add --map option to store memory map for specific tile
        target_link_options(${TILE_LIB} INTERFACE ${APP_LINK_OPTIONS} -Wm,--map,memory_tile${TILE}.ma)

        target_link_libraries(${TARGET_NAME} PUBLIC ${TILE_LIB})

        unset(TARGET_NAME)
        unset(TILE_LIB)
    endforeach()
    unset(LIB_NAME)

    #**********************
    # Merge binaries
    #**********************

    merge_binaries(
        "application_xvf3800_${TARGET_SUFFIX}"
        "tile0_application_xvf3800_${TARGET_SUFFIX}"
        "tile1_application_xvf3800_${TARGET_SUFFIX}"
        1)
    add_dependencies(all_ci "application_xvf3800_${TARGET_SUFFIX}")

    # Copy output to a handy location
    install(FILES "${CMAKE_CURRENT_BINARY_DIR}/application_xvf3800_${TARGET_SUFFIX}.xe" DESTINATION ${CMAKE_INSTALL_PREFIX}/../release-bin)

    # Copy output to convenient place as a build step.
    file(REAL_PATH ${CMAKE_BINARY_DIR}/../application_xvf3800_${TARGET_SUFFIX}.xe TARGET_COPY_PATH)
    file(RELATIVE_PATH TARGET_PRINT_PATH ${CMAKE_SOURCE_DIR} ${TARGET_COPY_PATH})
    add_custom_command(TARGET application_xvf3800_${TARGET_SUFFIX}
        POST_BUILD
        BYPRODUCTS ${TARGET_COPY_PATH}
        COMMAND ${CMAKE_COMMAND} -E copy_if_different ${CMAKE_CURRENT_BINARY_DIR}/application_xvf3800_${TARGET_SUFFIX}.xe ${TARGET_COPY_PATH}
        # print a message to make it clear that this binary has been built
        COMMAND ${CMAKE_COMMAND} -E cmake_echo_color --green --bold "Finished building ${TARGET_PRINT_PATH}")

    #**********************
    # Create run and debug targets
    #**********************
    add_custom_target("run_application_xvf3800_${TARGET_SUFFIX}"
      COMMAND "xrun --xscope application_xvf3800_${TARGET_SUFFIX}.xe"
      DEPENDS ${_EXECUTABLE_TARGET_NAME}
      COMMENT
        "Run application"
      VERBATIM
    )

    create_debug_target("application_xvf3800_${TARGET_SUFFIX}")
    create_flash_app_target("application_xvf3800_${TARGET_SUFFIX}")

endforeach()
